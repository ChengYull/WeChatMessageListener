微信群聊发言统计 App 完整设计方案
基于 Kotlin + Android 原生无障碍服务 的实现方案。通过监听微信聊天界面变化，提取发言人昵称，去重后写入本地 Room 数据库，并在 App 内展示发言排行榜。

StatsDao.kt


package com.example.wechatstats.data

import androidx.room.Dao
import androidx.room.Insert
import androidx.room.OnConflictStrategy
import androidx.room.Query
import kotlinx.coroutines.flow.Flow

@Dao
interface StatsDao {
    @Insert(onConflict = OnConflictStrategy.IGNORE)
    suspend fun insert(user: UserStats)

    @Query("UPDATE user_stats SET count = count + 1 WHERE nickname = :nickname")
    suspend fun incrementCount(nickname: String)

    @Query("SELECT * FROM user_stats ORDER BY count DESC")
    fun getAllStats(): Flow<List<UserStats>>
}
AppDatabase.kt


package com.example.wechatstats.data

import android.content.Context
import androidx.room.Database
import androidx.room.Room
import androidx.room.RoomDatabase

@Database(entities = [UserStats::class], version = 1)
abstract class AppDatabase : RoomDatabase() {
    abstract fun statsDao(): StatsDao

    companion object {
        @Volatile
        private var INSTANCE: AppDatabase? = null

        fun getDatabase(context: Context): AppDatabase {
            return INSTANCE ?: synchronized(this) {
                val instance = Room.databaseBuilder(
                    context.applicationContext,
                    AppDatabase::class.java,
                    "wechat_stats_db"
                ).build()
                INSTANCE = instance
                instance
            }
        }
    }
}
2. 无障碍服务 (WeChatMonitorService.kt)

package com.example.wechatstats

import android.accessibilityservice.AccessibilityService
import android.view.accessibility.AccessibilityEvent
import android.view.accessibility.AccessibilityNodeInfo
import com.example.wechatstats.data.AppDatabase
import kotlinx.coroutines.CoroutineScope
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.launch

class WeChatMonitorService : AccessibilityService() {

    private val serviceScope = CoroutineScope(Dispatchers.IO)
    private var lastProcessedMessage: String = "" // 用于简单的去重

    override fun onAccessibilityEvent(event: AccessibilityEvent?) {
        if (event?.packageName?.toString() != "com.tencent.mm") return

        val rootNode = rootInActiveWindow ?: return

        // 1. 查找聊天列表
        val chatListView = findChatListView(rootNode) ?: return

        // 2. 遍历最后 3 条消息（新消息通常在底部）
        val childCount = chatListView.childCount
        if (childCount == 0) return

        for (i in (childCount - 1) downTo maxOf(0, childCount - 3)) {
            val msgNode = chatListView.getChild(i) ?: continue

            // 3. 提取发言人
            val sender = extractSenderName(msgNode)

            if (sender != null && sender != lastProcessedMessage) {
                lastProcessedMessage = sender

                // 4. 写入数据库
                val dao = AppDatabase.getDatabase(applicationContext).statsDao()
                serviceScope.launch {
                    dao.insert(com.example.wechatstats.data.UserStats(sender, 0))
                    dao.incrementCount(sender)
                }
                break // 找到一条新消息就跳出，等下次触发
            }
        }
    }

    // 递归寻找 RecyclerView
    private fun findChatListView(root: AccessibilityNodeInfo): AccessibilityNodeInfo? {
        if (root.className == "androidx.recyclerview.widget.RecyclerView" ||
            root.className == "android.widget.ListView") {
            return root
        }
        for (i in 0 until root.childCount) {
            val child = root.getChild(i) ?: continue
            val found = findChatListView(child)
            if (found != null) return found
        }
        return null
    }

    // 提取发言人昵称
    private fun extractSenderName(msgNode: AccessibilityNodeInfo): String? {
        // 注意：这里的逻辑需要你用 Layout Inspector 实际查看微信结构！
        // 以下是一个通用的遍历查找逻辑，寻找包含文本的第一个 TextView
        // 在群聊中，别人的消息通常：头像 -> 昵称 TextView -> 消息气泡 TextView

        // 简单的遍历寻找法：找到第一个有文本的 TextView 视为昵称
        // （此逻辑较脆弱，强烈建议替换为 resource-id 查找）
        return findFirstTextNode(msgNode)
    }

    private fun findFirstTextNode(node: AccessibilityNodeInfo): String? {
        if (node.text != null && node.text.toString().isNotEmpty()) {
            val text = node.text.toString()
            // 过滤掉常见的系统提示或时间
            if (text.contains("撤回了一条消息") || text.contains(":") || text.length > 20) {
                return null
            }
            return text
        }
        for (i in 0 until node.childCount) {
            val child = node.getChild(i) ?: continue
            val result = findFirstTextNode(child)
            if (result != null) return result
        }
        return null
    }

    override fun onInterrupt() {}

    // 提供给外部检查服务是否运行
    companion object {
        var isRunning = false
    }

    override fun onServiceConnected() {
        super.onServiceConnected()
        isRunning = true
    }

    override fun onDestroy() {
        super.onDestroy()
        isRunning = false
    }
}
3. UI 层 (MainActivity.kt 与 StatsAdapter.kt)
StatsAdapter.kt


package com.example.wechatstats

import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import android.widget.TextView
import androidx.recyclerview.widget.DiffUtil
import androidx.recyclerview.widget.ListAdapter
import androidx.recyclerview.widget.RecyclerView
import com.example.wechatstats.data.UserStats

class StatsAdapter : ListAdapter<UserStats, StatsAdapter.StatsViewHolder>(DiffCallback()) {

    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): StatsViewHolder {
        val view = LayoutInflater.from(parent.context).inflate(R.layout.item_stats, parent, false)
        return StatsViewHolder(view)
    }

    override fun onBindViewHolder(holder: StatsViewHolder, position: Int) {
        val item = getItem(position)
        holder.tvNickname.text = "${position + 1}.${item.nickname}"
        holder.tvCount.text = "${item.count} 次"
    }

    class StatsViewHolder(view: View) : RecyclerView.ViewHolder(view) {
        val tvNickname: TextView = view.findViewById(R.id.tvNickname)
        val tvCount: TextView = view.findViewById(R.id.tvCount)
    }

    class DiffCallback : DiffUtil.ItemCallback<UserStats>() {
        override fun areItemsTheSame(oldItem: UserStats, newItem: UserStats) = oldItem.nickname == newItem.nickname
        override fun areContentsTheSame(oldItem: UserStats, newItem: UserStats) = oldItem == newItem
    }
}
MainActivity.kt


package com.example.wechatstats

import android.content.Intent
import android.os.Bundle
import android.provider.Settings
import android.view.WindowManager
import android.widget.Button
import androidx.appcompat.app.AppCompatActivity
import androidx.lifecycle.lifecycleScope
import androidx.recyclerview.widget.LinearLayoutManager
import androidx.recyclerview.widget.RecyclerView
import com.example.wechatstats.data.AppDatabase
import kotlinx.coroutines.flow.collectLatest
import kotlinx.coroutines.launch

class MainActivity : AppCompatActivity() {

    private lateinit var adapter: StatsAdapter

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        // 保持屏幕常亮，防止息屏导致无障碍失效
        window.addFlags(WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON)

        adapter = StatsAdapter()
        findViewById<RecyclerView>(R.id.recyclerViewStats).apply {
            layoutManager = LinearLayoutManager(this@MainActivity)
            this.adapter = this@MainActivity.adapter
        }

        // 跳转无障碍设置
        findViewById<Button>(R.id.btnOpenAccessibility).setOnClickListener {
            startActivity(Intent(Settings.ACTION_ACCESSIBILITY_SETTINGS))
        }

        // 监听数据库变化并更新 UI
        val dao = AppDatabase.getDatabase(this).statsDao()
        lifecycleScope.launch {
            dao.getAllStats().collectLatest { statsList ->
                adapter.submitList(statsList)
            }
        }
    }
}
六、 调试与避坑指南
完善 extractSenderName 方法（最重要）：
上述代码中的 findFirstTextNode 是一个比较粗糙的示例。你必须使用 Android Studio 的 Layout Inspector 连接手机，观察微信群聊的界面层级。
找到显示昵称的 TextView，查看它的 view id（例如可能是 com.tencent.mm:id/xxx）。
将 extractSenderName 替换为更精准的 ID 查找：

     val nameNodes = msgNode.findAccessibilityNodeInfosByViewId("com.tencent.mm:id/你的实际ID")
     if (nameNodes.isNotEmpty()) return nameNodes[0].text?.toString()
防杀后台：
在闲置手机的系统设置中，找到你的 App，允许“自启动”，并在电池设置中将其设为“无限制”或“不优化电池使用”。
挂机测试：
开启无障碍后，将微信停留在目标群聊界面。用另一个手机在群里发消息，观察你的 App 界面是否实时刷新了排行榜。
